Database Schema
device
id, name, exposure, latitude, longitude, model, grouptag, created_at, updated_at, last_measurement_at
sensor
id, device_id, title, unit, sensor_type, sensor_wiki_phenomenon
measurement
id, sensor_id, value, created_at
view_devices
(Simplified device view with sensor count)
view_measurements
(Measurements with device/sensor info)
Example Queries
SELECT * FROM device LIMIT 10;
SELECT name, latitude, longitude FROM device WHERE exposure = 'outdoor' LIMIT 50;
SELECT d.name, COUNT(s.id) as sensor_count FROM device d JOIN sensor s ON d.id = s.device_id GROUP BY d.id, d.name ORDER BY sensor_count DESC LIMIT 10;
SELECT title, unit, sensor_wiki_phenomenon FROM sensor WHERE sensor_wiki_phenomenon LIKE '%PM%' LIMIT 20;
SELECT * FROM view_devices LIMIT 20;
